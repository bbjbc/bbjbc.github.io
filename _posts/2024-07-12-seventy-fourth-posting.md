const graph = Array.from({ length: N + 1 }, () => []);

  const [a, b] = arr[i].split(" ").map(Number);
  graph[a].push(b);
  graph[b].push(a);
}

  graph[i].sort((a, b) => a - b);
}

function dfs(start) {
  const visited = Array(N + 1).fill(false);
  const result = [];

  function _dfs(v) {
    if (visited[v]) return;
    visited[v] = true;
    result.push(v);

    for (let next of graph[v]) {
      if (!visited[next]) {
        _dfs(next);
      }
    }
  }

  _dfs(start);
  return result;
}

function bfs(start) {
  const visited = Array(N + 1).fill(false);
  const result = [];
  const queue = [start];
  visited[start] = true;

  while (queue.length) {
    const node = queue.shift();
    result.push(node);

    for (let next of graph[node]) {
      if (!visited[next]) {
        visited[next] = true;
        queue.push(next);
      }
    }
  }

  return result;
}
