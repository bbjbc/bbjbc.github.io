// 0 0 1 0 3
// 0 2 5 0 1
// 4 2 4 4 2
// 3 5 1 3 1

function solution(board, moves) {
  let count = 0;
  const bucket = [];

  moves.forEach((move) => {
    for (let i = 0; i < board.length; i++) {
      if (board[i][move - 1]) {
        if (board[i][move - 1] !== bucket[bucket.length - 1]) {
          bucket.push(board[i][move - 1]);
        } else {
          bucket.pop();
          count += 2;
        }

        board[i][move - 1] = 0;
        break;
      }
    }
  });

  return count;
}
